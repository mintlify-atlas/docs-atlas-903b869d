## Overview

Dashi's frontend is built with React 19 and Next.js 15, using a component-based architecture with TypeScript for type safety. The UI features smooth animations powered by Framer Motion and a modular component structure.

## Component Hierarchy

```
App (layout.tsx)
└── CopilotKit Provider
    └── Home (page.tsx)
        └── AnimatedVectorBox (dashimain.tsx)
            ├── Dashi Avatar (SVG)
            ├── Music Notes Animation
            ├── Chat Interface
            └── Voice Response Display
```

#### Keyboard Event Handling

The component listens for keyboard shortcuts:

```typescript
useEffect(() => {
  const handleKeyDown = async (event: KeyboardEvent) => {
    // ESC - Reset to default
    if (event.key === "Escape") {
      handleReset();
      return;
    }

    // Cmd+Opt+T - Chat mode
    if ((event.metaKey || event.ctrlKey) && event.altKey && isTKey) {
      setIsActivatedState(true);
      setIsExpanded(false);
    }

    // Cmd+Opt+V - Voice mode
    if ((event.metaKey || event.ctrlKey) && event.altKey && isVKey) {
      setIsExpanded(true);
      await startRecording();
    }
  };

  const handleKeyUp = async (event: KeyboardEvent) => {
    // Release Cmd+Opt+V - Stop recording
    if ((event.metaKey || event.ctrlKey) && event.altKey && isVKey) {
      const result = await stopAndTranscribe();
      // Generate AI response
      const stream = await generate_response(result);
      // Update chat with streaming response
    }
  };

  document.addEventListener("keydown", handleKeyDown, true);
  document.addEventListener("keyup", handleKeyUp, true);
  return () => {
    document.removeEventListener("keydown", handleKeyDown, true);
    document.removeEventListener("keyup", handleKeyUp, true);
  };
}, [isActivated, isExpanded, isActivatedState, isRecording]);
```

## Custom Hooks

### useVoiceRecorder (`src/lib/tts.ts`)

Manages audio recording and transcription:

```typescript
export function useVoiceRecorder(): UseVoiceRecorderResult {
  const mediaRecorderRef = useRef<MediaRecorder | null>(null);
  const audioChunksRef = useRef<Blob[]>([]);
  const [isRecording, setIsRecording] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const startRecording = useCallback(async () => {
    const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
    const mediaRecorder = new MediaRecorder(stream, { mimeType: "audio/webm" });
    
    audioChunksRef.current = [];
    mediaRecorder.addEventListener("dataavailable", (event) => {
      audioChunksRef.current.push(event.data);
    });
    
    mediaRecorderRef.current = mediaRecorder;
    mediaRecorder.start();
    setIsRecording(true);
  }, []);

  const stopAndTranscribe = useCallback(async (): Promise<string | null> => {
    return new Promise((resolve, reject) => {
      mediaRecorderRef.current!.addEventListener("stop", async () => {
        // Create audio blob
        const audioBlob = new Blob(audioChunksRef.current);
        
        // Convert to PCM format
        const audioContext = new AudioContext({ sampleRate: 16000 });
        const arrayBuffer = await audioBlob.arrayBuffer();
        const audioBuffer = await audioContext.decodeAudioData(arrayBuffer);
        const pcmData = audioBuffer.getChannelData(0);
        const audioSamples = Array.from(pcmData);
        
        // Send to Tauri backend
        const transcription = await invoke<string>("transcribe", {
          audio: audioSamples,
        });
        
        resolve(transcription);
      }, { once: true });
      
      mediaRecorderRef.current!.stop();
    });
  }, []);

  return { isRecording, error, startRecording, stopAndTranscribe };
}
```

**Key Features:**
- Uses `MediaRecorder` API to capture microphone audio
- Converts WebM audio to PCM format at 16kHz
- Invokes Tauri backend via `invoke()` for transcription
- Returns transcribed text as a string

## UI Component Library

Dashi uses Radix UI primitives with custom styling:

### Button Component (`src/components/ui/button.tsx`)

```typescript
import * as React from "react";
import { Slot } from "@radix-ui/react-slot";
import { cva, type VariantProps } from "class-variance-authority";

const buttonVariants = cva(
  "inline-flex items-center justify-center rounded-md transition-colors",
  {
    variants: {
      variant: {
        default: "bg-primary text-primary-foreground hover:bg-primary/90",
        outline: "border border-input hover:bg-accent",
      },
      size: {
        default: "h-10 px-4 py-2",
        sm: "h-9 px-3",
        lg: "h-11 px-8",
      },
    },
    defaultVariants: {
      variant: "default",
      size: "default",
    },
  }
);

export interface ButtonProps
  extends React.ButtonHTMLAttributes<HTMLButtonElement>,
    VariantProps<typeof buttonVariants> {
  asChild?: boolean;
}

const Button = React.forwardRef<HTMLButtonElement, ButtonProps>(
  ({ className, variant, size, asChild = false, ...props }, ref) => {
    const Comp = asChild ? Slot : "button";
    return (
      <Comp
        className={cn(buttonVariants({ variant, size, className }))}
        ref={ref}
        {...props}
      />
    );
  }
);
```
